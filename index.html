        CSS (styles.css) body { font-family: Arial, sans-serif; } .course-container, .college-container, .duration-container, .job-container { margin-bottom: 20px; } #courses-list, #colleges-list, #job-opportunities-list { list-style: none; padding: 0; margin: 0; } #courses-list li, #colleges-list li, #job-opportunities-list li { padding: 10px; border-bottom: 1px solid #ccc; } JavaScript (script.js) // Sample data const courses = [ { name: "BCA", colleges: ["Aditya Degree College", "Kurnool University"], duration: "3 years", jobOpportunities: ["Software Developer", "Data Analyst"] }, { name: "BSc", colleges: ["Kurnool University", "Rayalaseema University"], duration: "3 years", jobOpportunities: ["Research Scientist", "Lab Assistant"] } ]; // Populate courses list const coursesList = document.getElementById("courses-list"); courses.forEach(course => { const li = document.createElement("li"); li.textContent = course.name; coursesList.appendChild(li); }); // Populate colleges list based on selected course const collegesList = document.getElementById("colleges-list"); const courseDuration = document.getElementById("course-duration"); const jobOpportunitiesList = document.getElementById("job-opportunities-list"); coursesList.addEventListener("click", (e) => { if (e.target.tagName === "LI") { const selectedCourse = courses.find(course => course.name === e.target.textContent); collegesList.innerHTML = ""; selectedCourse.colleges.forEach(college => { const li = document.createElement("li"); li.textContent = college; collegesList.appendChild(li); }); courseDuration.textContent = `Duration: ${selectedCourse.duration}`; jobOpportunitiesList.innerHTML = ""; selectedCourse.jobOpportunities.forEach(job => { const li = document.createElement("li"); li.textContent = job; jobOpportunitiesList.appendChild(li); }); } });
